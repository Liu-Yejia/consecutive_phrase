#输入一条各元素为字符串的向量，将它们按顺序粘起来之后输出。
fun_InVec_adhere<-function(vec, sep="")
{
  library(stringr)
  len<-length(vec)
  if(len>=2){
    res<-vec[1]
    for(i in 2:len){
      if(vec[i]!=""){
        res<-str_c(res, vec[i], sep=sep)
      }else{
        break
      }
    }
  }else{
    res<-vec
  }
  return(res)
}#对向量；矩阵黏合函数，自动检查输入为向量还是矩阵，若输入为矩阵的话，调用direction参数；
